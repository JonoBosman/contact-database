<script>
// ─── CONFIG ───────────────────────────────────────────────────────────────────
const COLUMNS = [
  { key: "name",        label: "First Name" },
  { key: "surname",     label: "Last Name" },
  { key: "organisation",label: "Organisation" },
  { key: "source_db",   label: "Source Database", type: "select" },
  { key: "email",       label: "Email" },
  { key: "phone",       label: "Phone" },
  { key: "job_title",   label: "Job Title" },
  { key: "country",     label: "Country" },
  { key: "date_added",  label: "Date Added" },
  { key: "tags",        label: "Tags" },
  { key: "notes",       label: "Notes" },
];
const DB_OPTIONS = ["CRM","LinkedIn","HubSpot","Salesforce","Mailchimp","Manual Entry","Imported CSV","Zoho","Pipedrive","Other"];
const DB_COLORS  = { CRM:"#3b82f6", LinkedIn:"#0077b5", HubSpot:"#f97316", Salesforce:"#00a1e0", Mailchimp:"#b45309", "Manual Entry":"#8b5cf6", "Imported CSV":"#10b981", Zoho:"#e53e3e", Pipedrive:"#26292c", Other:"#6b7280" };

// ─── STATE ────────────────────────────────────────────────────────────────────
let rows        = [];
let visibleCols = COLUMNS.map(c => c.key);
let sortConfig  = { key: null, dir: "asc" };
let editCell    = null; // { rowIdx, colKey }
let scriptUrl   = localStorage.getItem("gs_script_url") || "";

// ─── INIT ─────────────────────────────────────────────────────────────────────
window.onload = () => {
  buildColMenu();
  buildHeader();
  if (scriptUrl) {
    document.getElementById("setup-banner").style.display = "none";
    document.getElementById("script-url-input").value = scriptUrl;
    loadFromSheet();
  } else {
    renderTable();
  }
};

function saveScriptUrl() {
  const val = document.getElementById("script-url-input").value.trim();
  if (!val.startsWith("https://script.google.com")) {
    showStatus("That doesn't look like a valid Apps Script URL. Please check and try again.", "err"); return;
  }
  scriptUrl = val;
  localStorage.setItem("gs_script_url", scriptUrl);
  document.getElementById("setup-banner").style.display = "none";
  showStatus("Connected! Loading data from Google Sheet…", "info");
  loadFromSheet();
}

// ─── GOOGLE SHEETS API ────────────────────────────────────────────────────────
async function loadFromSheet() {
  if (!scriptUrl) return;
  showStatus("Loading data from Google Sheet…", "info");
  try {
    const res  = await fetch(scriptUrl + "?action=get");
    const data = await res.json();
    if (data.error) throw new Error(data.error);
    rows = data.rows || [];
    renderTable();
    showStatus("✅ Data loaded successfully.", "ok");
  } catch(e) {
    showStatus("❌ Could not load from Google Sheet: " + e.message, "err");
  }
}

async function saveToSheet(row, rowIdx) {
  if (!scriptUrl) return;
  showStatus("Saving…", "info");
  try {
    const res  = await fetch(scriptUrl, {
      method: "POST",
      body: JSON.stringify({ action: "upsert", row, rowIdx })
    });
    const data = await res.json();
    if (data.error) throw new Error(data.error);
    showStatus("✅ Saved to Google Sheet.", "ok");
  } catch(e) {
    showStatus("❌ Save failed: " + e.message, "err");
  }
}

async function deleteFromSheet(rowIdx) {
  if (!scriptUrl) return;
  showStatus("Deleting…", "info");
  try {
    const res  = await fetch(scriptUrl, {
      method: "POST",
      body: JSON.stringify({ action: "delete", rowIdx })
    });
    const data = await res.json();
    if (data.error) throw new Error(data.error);
    showStatus("✅ Row deleted.", "ok");
  } catch(e) {
    showStatus("❌ Delete failed: " + e.message, "err");
  }
}

// ─── TABLE RENDER ─────────────────────────────────────────────────────────────
function buildHeader() {
  const cols = COLUMNS.filter(c => visibleCols.includes(c.key));
  const tr   = document.createElement("tr");
  cols.forEach(c => {
    const th = document.createElement("th");
    const arrow = sortConfig.key === c.key ? (sortConfig.dir === "asc" ? " ▲" : " ▼") : " ⇅";
    th.innerHTML = c.label + `<span style="opacity:.4">${arrow}</span>`;
    th.onclick = () => { sortConfig = { key: c.key, dir: sortConfig.key === c.key && sortConfig.dir === "asc" ? "desc" : "asc" }; buildHeader(); renderTable(); };
    tr.appendChild(th);
  });
  const th = document.createElement("th"); th.textContent = "Actions"; tr.appendChild(th);
  const thead = document.getElementById("thead"); thead.innerHTML = ""; thead.appendChild(tr);
}

function renderTable() {
  const search = document.getElementById("search-input").value.toLowerCase();
  const cols   = COLUMNS.filter(c => visibleCols.includes(c.key));
  let filtered = rows.filter(row => cols.some(c => String(row[c.key]||"").toLowerCase().includes(search)));
  if (sortConfig.key) {
    filtered.sort((a,b) => {
      const av = a[sortConfig.key]||"", bv = b[sortConfig.key]||"";
      return sortConfig.dir === "asc" ? av.localeCompare(bv) : bv.localeCompare(av);
    });
  }
  document.getElementById("record-count").textContent = rows.length + " record" + (rows.length !== 1 ? "s" : "");
  const tbody = document.getElementById("tbody"); tbody.innerHTML = "";
  if (filtered.length === 0) {
    const tr = document.createElement("tr");
    const td = document.createElement("td"); td.colSpan = cols.length + 1;
    td.style.cssText = "text-align:center;padding:30px;color:#94a3b8";
    td.textContent = "No records found."; tr.appendChild(td); tbody.appendChild(tr); return;
  }
  filtered.forEach((row) => {
    const origIdx = rows.indexOf(row);
    const tr = document.createElement("tr");
    cols.forEach(c => {
      const td = document.createElement("td");
      if (editCell && editCell.rowIdx === origIdx && editCell.colKey === c.key) {
        if (c.type === "select") {
          const sel = document.createElement("select");
          sel.className = "cell";
          sel.innerHTML = `<option value="">-- Select --</option>` + DB_OPTIONS.map(o => `<option${row[c.key]===o?" selected":""}>${o}</option>`).join("");
          sel.onchange = () => commitEdit(origIdx, c.key, sel.value);
          sel.onblur   = () => commitEdit(origIdx, c.key, sel.value);
          td.appendChild(sel); setTimeout(()=>sel.focus(),0);
        } else {
          const inp = document.createElement("input");
          inp.className = "cell"; inp.value = row[c.key] || "";
          inp.onblur   = () => commitEdit(origIdx, c.key, inp.value);
          inp.onkeydown = e => { if(e.key==="Enter") commitEdit(origIdx, c.key, inp.value); };
          td.appendChild(inp); setTimeout(()=>inp.focus(),0);
        }
      } else {
        const div = document.createElement("div");
        div.className = "cell" + (row[c.key] ? "" : " empty");
        if (c.key === "source_db" && row[c.key]) {
          const span = document.createElement("span");
          span.className = "badge";
          span.style.background = DB_COLORS[row[c.key]] || "#6b7280";
          span.textContent = row[c.key];
          div.appendChild(span);
        } else {
          div.textContent = row[c.key] || "click to edit";
        }
        div.onclick = () => { editCell = { rowIdx: origIdx, colKey: c.key }; renderTable(); };
        td.appendChild(div);
      }
      tr.appendChild(td);
    });
    // Delete button
    const tdDel = document.createElement("td"); tdDel.style.cssText = "padding:6px 10px;text-align:center";
    const btn   = document.createElement("button"); btn.className = "btn btn-red"; btn.textContent = "🗑";
    btn.title = "Delete row";
    btn.onclick = () => {
      if (!confirm("Delete this row?")) return;
      rows.splice(origIdx, 1);
      deleteFromSheet(origIdx);
      renderTable();
    };
    tdDel.appendChild(btn); tr.appendChild(tdDel);
    tbody.appendChild(tr);
  });
}

function commitEdit(rowIdx, colKey, val) {
  rows[rowIdx][colKey] = val;
  editCell = null;
  saveToSheet(rows[rowIdx], rowIdx);
  renderTable();
}

// ─── ADD ROW ──────────────────────────────────────────────────────────────────
function addRow() {
  const row = Object.fromEntries(COLUMNS.map(c => [c.key, ""]));
  rows.push(row);
  saveToSheet(row, rows.length - 1);
  renderTable();
}

// ─── COLUMN TOGGLE ────────────────────────────────────────────────────────────
function buildColMenu() {
  const menu = document.getElementById("col-menu"); menu.innerHTML = "";
  const title = document.createElement("div"); title.className = "col-menu-title"; title.textContent = "TOGGLE COLUMNS"; menu.appendChild(title);
  COLUMNS.forEach(c => {
    const lbl = document.createElement("label");
    const chk = document.createElement("input"); chk.type = "checkbox"; chk.checked = visibleCols.includes(c.key);
    chk.onchange = () => { visibleCols = chk.checked ? [...visibleCols, c.key] : visibleCols.filter(k => k !== c.key); buildHeader(); renderTable(); };
    lbl.appendChild(chk); lbl.append(" " + c.label); menu.appendChild(lbl);
  });
}
function toggleColMenu() {
  const m = document.getElementById("col-menu");
  m.style.display = m.style.display === "none" ? "block" : "none";
}
document.addEventListener("click", e => { if (!e.target.closest(".col-wrap")) document.getElementById("col-menu").style.display = "none"; });

// ─── EXPORT CSV ───────────────────────────────────────────────────────────────
function exportCSV() {
  const cols = COLUMNS.filter(c => visibleCols.includes(c.key));
  const lines = [cols.map(c => `"${c.label}"`).join(","),
    ...rows.map(row => cols.map(c => `"${String(row[c.key]||"").replace(/"/g,'""')}"`).join(","))];
  const blob = new Blob([lines.join("\n")], { type: "text/csv;charset=utf-8;" });
  const a = document.createElement("a"); a.href = URL.createObjectURL(blob);
  a.download = "contacts.csv"; a.click();
}

// ─── IMPORT CSV ───────────────────────────────────────────────────────────────
function importCSV(e) {
  const file = e.target.files[0]; if (!file) return;
  const reader = new FileReader();
  reader.onload = async ev => {
    const lines = ev.target.result.trim().split("\n").map(l=>l.trim()).filter(Boolean);
    if (lines.length < 2) { alert("CSV appears empty."); return; }
    const parseRow = line => {
      const cells=[]; let cur="", inQ=false;
      for (let i=0;i<line.length;i++) {
        const ch=line[i];
        if(ch==='"'&&!inQ){inQ=true;}
        else if(ch==='"'&&inQ&&line[i+1]==='"'){cur+='"';i++;}
        else if(ch==='"'&&inQ){inQ=false;}
        else if(ch===','&&!inQ){cells.push(cur.trim());cur="";}
        else cur+=ch;
      } cells.push(cur.trim()); return cells;
    };
    const headers = parseRow(lines[0]).map(h=>h.toLowerCase().replace(/\s+/g,"_").replace(/[^a-z0-9_]/g,""));
    const colKeys = COLUMNS.map(c=>c.key);
    const imported = lines.slice(1).map(line => {
      const cells = parseRow(line);
      const row = Object.fromEntries(COLUMNS.map(c=>[c.key,""]));
      headers.forEach((h,i) => {
        const match = colKeys.find(k=>k===h||COLUMNS.find(c=>c.key===k)?.label.toLowerCase().replace(/\s+/g,"_")===h);
        if (match) row[match] = cells[i]||"";
      });
      return row;
    });
    rows.push(...imported);
    // Save all imported rows to sheet
    for (let i=0;i<imported.length;i++) await saveToSheet(imported[i], rows.length - imported.length + i);
    renderTable();
    showStatus(`✅ Imported ${imported.length} row(s) successfully.`, "ok");
    e.target.value = "";
  };
  reader.readAsText(file);
}

// ─── STATUS BAR ───────────────────────────────────────────────────────────────
let statusTimer;
function showStatus(msg, type) {
  const el = document.getElementById("status");
  el.textContent = msg; el.className = "status-bar show " + type;
  clearTimeout(statusTimer);
  statusTimer = setTimeout(() => el.className = "status-bar", 4000);
}
